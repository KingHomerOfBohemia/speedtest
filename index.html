Měření Downloadu (20MB soubor) // Používáme náhodný parametr v URL, aby se obešla mezipaměť (cache) const downloadUrl = "https://speed.cloudflare.com/__down?bytes=20000000"; const startTime = Date.now(); let loadedBytes = 0; let speedHistory = []; try { const response = await fetch(downloadUrl, { cache: 'no-store' }); const reader = response.body.getReader(); while (true) { const { done, value } = await reader.read(); if (done) break; loadedBytes += value.length; const durationInSeconds = (Date.now() - startTime) / 1000; const bps = (loadedBytes * 8) / durationInSeconds; const mbps = (bps / 1000000).toFixed(2); downloadDisplay.innerText = `${mbps} Mbps`; // Aktualizace grafu speedHistory.push(mbps); if (speedHistory.length > 20) speedHistory.shift(); speedChart.data.datasets[0].data = speedHistory; speedChart.update('none'); } } catch (e) { downloadDisplay.innerText = "Chyba stahování"; } // 3.
